# 낚시왕



```cpp
#include <iostream>
#include <vector>
#include <map>
#include <cstring>
using namespace std;

struct Shark
{
	int r;
	int c;
	int s; // 속력
	int d; // 이동
	int z; // 크기
};

int main() {
	int R, C, M;
	int r, c, s, d, z;
	map<int, Shark> sharkmap; // 상어정보들 [크기, 정보구조체]
	map<int, Shark>::iterator it;
	int row[5] = {0, -1, 1, 0, 0}; // 1:위, 2:아래, 3: 오른쪽, 4:왼쪽
	int col[5] = {0, 0, 0, 1, -1};
	int ans = 0;
	
	scanf("%d %d %d", &R, &C, &M);
	int arr[R+1][C+1] = {0};
	int arrtmp[R+1][C+1] = {0};
	for (int i=0; i<M; i++) {
		scanf("%d %d %d %d %d", &r, &c, &s, &d, &z);
		arr[r][c] = z;
		Shark tmp = {r, c, s, d, z};
		sharkmap.insert(make_pair(z, tmp));
	}
	
	
	int nr, nc, nd, move;
	for (int i=1; i<=C; i++) {
		
		
		// 상어잡기
		for (int j=1; j<=R; j++) {
			if (arr[j][i] > 0) {
				sharkmap.erase(arr[j][i]);
				ans += arr[j][i];
				arr[j][i] = 0;
				break;
			}
		}
		
		// 상어이동
		for(it=sharkmap.begin(); it != sharkmap.end(); it++) {
			Shark &shark = it -> second;
			// 오른쪽, 왼쪽
			if (shark.d == 3 || shark.d == 4) {
				move = shark.s % ((C-1)*2);	// 실제 이동해야할 MOVE
				nc = shark.c;
				nd = shark.d;
				while (move > 0) {
					if (nc == 1) nd = 3;
					else if (nc == C) nd = 4;
					nc += col[nd];
					move--;
				}
				shark.d = nd;
				shark.c = nc;
			} 
			// 위, 아래 
			else if (shark.d == 1 || shark.d == 2) {
				move = shark.s % ((R-1)*2);	// 실제 이동해야할 MOVE
				nr = shark.r;
				nd = shark.d;
				while (move > 0) {
					if (nr == 1) nd = 2;
					else if (nr == R) nd = 1;
					nr += row[nd];
					move--;
				}
				shark.d = nd;
				shark.r = nr;
			}
			// 상어 먹기
			if (arrtmp[shark.r][shark.c] < shark.z) {
				if (arrtmp[shark.r][shark.c] > 0) 
					sharkmap.erase(arrtmp[shark.r][shark.c]);
				
				arrtmp[shark.r][shark.c] = shark.z;
					
			} else sharkmap.erase(shark.z);
			
		
		}
		
		for (int j=1; j<=R; j++) {
			for (int k=1; k<=C; k++) {
				arr[j][k] = arrtmp[j][k];
				arrtmp[j][k] = 0;
			}
		}
	}
	
	printf("%d", ans);
	
	return 0;
}
```
